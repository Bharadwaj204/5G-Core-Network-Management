' PowerPoint VBA Template for 5G Core Management Presentation
' This is a template that demonstrates how to create slides programmatically
' You will need to customize this with your specific content

Sub Create5GCorePresentation()
    Dim pptApp As PowerPoint.Application
    Dim pptPres As PowerPoint.Presentation
    Dim pptSlide As PowerPoint.slide
    Dim pptLayout As PowerPoint.CustomLayout
    Dim slideIndex As Integer
    
    ' Create PowerPoint application
    Set pptApp = New PowerPoint.Application
    pptApp.Visible = True
    
    ' Create new presentation
    Set pptPres = pptApp.Presentations.Add
    
    ' Slide 1: Title Slide
    slideIndex = pptPres.Slides.Count + 1
    Set pptSlide = pptPres.Slides.Add(slideIndex, ppLayoutTitle)
    With pptSlide
        .Shapes.Title.TextFrame.TextRange.Text = "End-to-End 5G Core Management Prototype"
        .Shapes.Placeholders(2).TextFrame.TextRange.Text = "Your Name" & vbCrLf & "Date"
    End With
    
    ' Slide 2: Problem Statement
    slideIndex = pptPres.Slides.Count + 1
    Set pptSlide = pptPres.Slides.Add(slideIndex, ppLayoutTitleAndContent)
    With pptSlide
        .Shapes.Title.TextFrame.TextRange.Text = "Problem Statement"
        .Shapes.Placeholders(2).TextFrame.TextRange.Text = _
            "• Students have theoretical knowledge of 5G Core, but no hands-on platform to learn management" & vbCrLf & _
            "• Real telecom systems are expensive and complex" & vbCrLf & _
            "• Need a lightweight, open-source, manageable 5G Core"
    End With
    
    ' Slide 3: Project Objective
    slideIndex = pptPres.Slides.Count + 1
    Set pptSlide = pptPres.Slides.Add(slideIndex, ppLayoutTitleAndContent)
    With pptSlide
        .Shapes.Title.TextFrame.TextRange.Text = "Project Objective"
        .Shapes.Placeholders(2).TextFrame.TextRange.Text = _
            "• Deploy Open5GS (a free 5G Core implementation)" & vbCrLf & _
            "• Manage it using industry-standard protocols" & vbCrLf & _
            "  - NETCONF" & vbCrLf & _
            "  - RESTCONF" & vbCrLf & _
            "  - SNMP" & vbCrLf & _
            "• Build YANG models" & vbCrLf & _
            "• Create dashboards and monitoring" & vbCrLf & _
            "• Provide easy test scripts"
    End With
    
    ' Slide 4: 5G Core Architecture
    slideIndex = pptPres.Slides.Count + 1
    Set pptSlide = pptPres.Slides.Add(slideIndex, ppLayoutTitleAndContent)
    With pptSlide
        .Shapes.Title.TextFrame.TextRange.Text = "5G Core Architecture"
        .Shapes.Placeholders(2).TextFrame.TextRange.Text = _
            "• AMF (Access and Mobility Management)" & vbCrLf & _
            "• SMF (Session Management)" & vbCrLf & _
            "• UPF (User Plane)" & vbCrLf & _
            "• NRF, UDM, AUSF" & vbCrLf & _
            "• N2/N3 interfaces" & vbCrLf & _
            "• UE/gNB (optional)"
    End With
    
    ' Slide 5: System Architecture
    slideIndex = pptPres.Slides.Count + 1
    Set pptSlide = pptPres.Slides.Add(slideIndex, ppLayoutTitleAndContent)
    With pptSlide
        .Shapes.Title.TextFrame.TextRange.Text = "What We Built (System Architecture)"
        .Shapes.Placeholders(2).TextFrame.TextRange.Text = _
            "Two main layers:" & vbCrLf & vbCrLf & _
            "1. 5G Core Plane (Open5GS: AMF/SMF/UPF/etc.)" & vbCrLf & vbCrLf & _
            "2. Management Plane:" & vbCrLf & _
            "   • NETCONF Server" & vbCrLf & _
            "   • RESTCONF API" & vbCrLf & _
            "   • SNMP Monitor" & vbCrLf & _
            "   • YANG Models" & vbCrLf & _
            "   • Dashboard"
    End With
    
    ' Add more slides as needed...
    
    ' Save the presentation
    pptPres.SaveAs "C:\Users\91939\Desktop\nmspro\5g-core-management\docs\5G-Core-Management-Presentation.pptx"
    
    MsgBox "Presentation created successfully!", vbInformation
End Sub

' Helper function to add content to slides
Sub AddSlideWithContent(slideTitle As String, slideContent As String)
    Dim pptApp As PowerPoint.Application
    Dim pptPres As PowerPoint.Presentation
    Dim pptSlide As PowerPoint.slide
    Dim slideIndex As Integer
    
    ' Get the active presentation
    Set pptApp = PowerPoint.Application
    Set pptPres = pptApp.ActivePresentation
    
    ' Add new slide
    slideIndex = pptPres.Slides.Count + 1
    Set pptSlide = pptPres.Slides.Add(slideIndex, ppLayoutTitleAndContent)
    
    With pptSlide
        .Shapes.Title.TextFrame.TextRange.Text = slideTitle
        .Shapes.Placeholders(2).TextFrame.TextRange.Text = slideContent
    End With
End Sub
